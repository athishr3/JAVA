class Main {
    
    public static void main(String[] args) {
        int arr[][] = { {1,4,5,8,}, {11,12,13,14}};
        for(int i=0;i<arr.length;i++){
          for(int j=0;j<arr.length;j++)
          {
            System.out.print(arr[i][j]+" ");
        }
        System.out.println(" ");
    }
}
}
